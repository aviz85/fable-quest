## 3. Puzzle script — onCommand

סה"כ ניקוד חדר: **30** = 10+10+10.

```
// --- מעטפה ---
(verb 'קח' && noun 'מעטפה'):
  if hasItem('maatefet') → say("כבר לקחת אותה. היא בכיס. כמו הצרות."); return true
  addItem('maatefet', 'מעטפה מסווגת',
    "חותמת 'סודי ביותר' וכתם חומוס. הקואורדינטות בפנים נראות לך מוכרות באופן מטריד.")
  score(10, 'cafe_main_envelope')
  say("אתה שולף את המעטפה מתחת לזכוכית. ארבעה תאגידים רוצים אותה, ואתה רוצה פיקדון.\nמישהו כאן יצא פראייר, ועוד לא ברור מי.")
  return true

(verb 'קרא'|'פתח' && noun 'מעטפה'):
  if !hasItem('maatefet') → say("קודם תרים אותה. אתה אב בית, לא קורא מחשבות."); return true
  say("בפנים: שורת קואורדינטות. 32.0700 צפון, 34.7749 מזרח.\nמשהו בהן מגרד לך במוח, כמו שם של שכן שאתה לא זוכר.\nהקריין היה יכול להגיד לך מה זה. הקריין בוחר שלא. מטעמי... דרמה.")
  return true

// --- מכונת הקפה (בעיטה) ---
(verb 'דחוף'|'לחץ'|'משוך'|'השתמש' && noun 'מכונת קפה'):
  say("לחצת על הכפתור בנימוס. המכונה מזלזלת בנימוס. נסה שפה שהיא מבינה — בעיטה.")
  return true

(verb 'בעט' && noun 'מכונת קפה'):   // 'בעט' מגיע כפועל גולמי; לזהות גם raw "בעט במכונה"
  if getFlag('cafe_coffee_done') →
    say("בעטת שוב. המכונה משמיעה חרחור של 'זהו, זה היה הקפה האחרון שלי' ונכבית לצמיתות.\nכל הכבוד. הרגת את הדבר האחרון שעבד פה."); return true
  setFlag('cafe_coffee_done')
  addItem('kafe_botz', 'כוס קפה בוץ',
    'הקפה היחיד באזור שלא עבר ועדת בטיחות קפאין. מריח כמו החלטה גרועה.')
  score(10, 'cafe_main_coffee')
  say("בעיטה מדויקת של ארבעים שנות ניסיון. המכונה נאנחת, רועדת, ופולטת כוס אחת של בוץ רותח.\nעל הכוס כתוב 'העובד המצטיין 1998'. זה לא היית אתה. זה כואב עד היום.")
  return true

(verb 'שתה' && noun-item 'kafe_botz' or noun 'קפה'):
  say("כמעט. שפתיך נגעו בכוס, ואז המוח שלך — בפעולת חירום נדירה — הזכיר לך שמישהו במנזר\nישלם על זה בחתימה. אתה מוריד את הכוס. הקפאין בוכה.")
  return true

// --- הטלפון (מוקד המלחמה) ---
(verb 'השתמש'|'דבר'|'הרם'|'לחץ' && noun 'טלפון'):
  if hasItem('mispar_pnia') →
    say("חייגת שוב. 'זיהינו שכבר יש לך מספר פנייה. שיחתך חשובה לנו פחות. להתראות.'"); return true
  say("אתה מחייג למוקד המלחמה הבין-תאגידי. שלוחה 9.\n'שיחתך חשובה לנו. מיקומך בתור: 4,096.'")
  say("נעימת ההמתנה: גרסת פאנפלוט ל'עוד אברבנאל'. אתה מרגיש את נשמתך עוזבת בשלווה.")
  say("'זמן ההמתנה המשוער: רבעון פיסקאלי.' ...ואז נס: 'עקב המלחמה, כל נציגינו פוטרו.\nמונפק לך מספר פנייה אוטומטי. תודה שנלחמת איתנו.'")
  addItem('mispar_pnia', 'פתק עם מספר פנייה',
    'מספר פנייה 4096 מהמוקד הבין-תאגידי. שווה זהב בפני כל ועדה.')
  score(10, 'cafe_main_hotline')
  say("רשמת את המספר על פתק. 4096. מספר עגול להפליא, למי שסופר בבינארי.\nאתה לא סופר בבינארי. אתה סופר בפיקדונות.")
  return true

// --- מוות: טיפוס דרך החלון ---
(verb 'טפס'|'עבור'|'לך' && noun 'חלון'):
  → die (§5)

// --- רמז מגונן (זריעת הטוויסט) ---
(verb 'הסתכל' && noun 'מדרגות'):  // בנוסף ל-look בטבלה, פעם ראשונה בלבד (flag cafe_stairs_hint)
  say("הקריין מבקש, באופן חריג ואישי: עזוב את המרתף. השרת שם... רגיש.\n...בעצם, לך. רד. מה אכפת לי. אני בסך הכול קול.")
```

## 4. Dialogue — onEnter וסצנת הפתיחה

```
onEnter(api):
  if !getFlag('intro_done'):
    setFlag('intro_done'); setFlag('visited_cafe_main')
    say("אינטרנט קפה 'שוקי 2000', מרכז תל אביב, 2027. שלוש שנים לתוך מלחמת ה-AI הגדולה,\nואתה — יענקל'ה סרברמן, אב בית — עדיין מחכה לפיקדון על החלון. על החלון ההוא. שנשבר. עכשיו.")
    say("בום. רחפן שליחויות מתרסק דרך החלון בסגנון של מישהו שלא קרא את תנאי השימוש של פיזיקה.\nמעטפה צהובה עפה ממנו ונוחתת מתחת לערימת זכוכית. הרחפן מהבהב באדום, בעיקר מעלבון.")
    say("ארבע דמויות בחליפות פורצות פנימה בו-זמנית: נזיר, איש קינוט, בוט מעשן, ומנהלת מוצר.\nהם מסתכלים עליך. על המעטפה. עליך. ואז, פה-אחד — ההסכמה הראשונה שלהם מאז 2024:")
    say("'זה הוא. הסוכן פייבל.'\nוהם נעלמים לדווח לממונים. אתה נשאר עם מגב רגשי, חלון שבור, ומוניטין של סוכן-על.\nהפיקדון, אגב? עוד לא חזר.")
  else if !getFlag('visited_cafe_main'):
    setFlag('visited_cafe_main')
```

- כניסות חוזרות (אחרי הדגלים): שקט. ה-description הקבוע: "בית הקפה של שוקי: שלושה מחשבים מ-2003, מכונת קפה מ-2019, וחור בצורת רחפן בחלון. אתה עובד פה. מישהו צריך."
- אין NPC פעיל קבוע — הרחפן עונה ל'דבר' (טבלת hotspots), ארבעת הסוכנים קיימים רק בסצנת הפתיחה.
