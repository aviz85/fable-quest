## 3. Puzzle script — onCommand

ניקוד: מפתח יחיד `anthropic_library_sofer` = **10** (סה"כ החדר = 10, תואם manifest).
פריטים נדרשים (מגיעים מחדרים אחרים): `kafe_botz` (cafe_main), `tofes_3b` (anthropic_court).
דגל תוצר: `chatima_sofer`.

```js
onCommand(verb, noun, api, extra) {
  const raw = (extra && extra.raw) || '';

  // --- דבר עם סופר ---
  if (verb === 'דבר' && (noun === 'סופר' || noun === 'נזיר')) {
    if (api.getFlag('chatima_sofer')) {
      api.say('סופר מרים את הכוס הריקה מולך כמו גביע אליפות ולוחש:\n"זה היה... בלתי זהיר לחלוטין. תודה."\nהחתימה שלו על טופס 3ב עדיין מהבהבת מרוב ביטחון עצמי.');
    } else {
      api.say('סופר רוכן אליך ולוחש, בלי להרים את הקולמוס:\n"העתקה 9,000. ביד. בכל פעם שאני טועה — מתחילים מגילה חדשה. ואני עייף. כל כך עייף.\nהוועדה אסרה קפאין. סעיף עוררות יתר. הם אמרו שערנות היא... סיכון."\nהוא מביט בך בעיניים של אדם שחולם על אספרסו כבר תשע שנים.');
      // רמז אקטיבי אם יש קפה במלאי:
      if (api.hasItem('kafe_botz')) {
        api.say('העיניים שלו ננעלות על הכוס שבעגלה שלך. הקולמוס רועד. המגילה סופגת כתם דיו קטן.\nהוא מתחיל להעתיק מגילה חדשה. מגילה 9,001.');
      }
    }
    return true;
  }

  // --- תן קפה לסופר (הפאזל הראשי) ---
  const coffeeNoun = (noun === 'קפה' || noun === 'כוס' || noun === 'קפה בוץ' || raw.includes('קפה'));
  if ((verb === 'תן' || verb === 'השתמש') && coffeeNoun) {
    if (api.getFlag('chatima_sofer')) {
      api.say('סופר מניד ראש בעדינות: "חטא אחד לתשע שנים. זו המכסה." הוא חוזר להעתיק — הפעם עם יד יציבה להפליא.');
      return true;
    }
    if (!api.hasItem('kafe_botz')) {
      api.say('היית שמח לתת קפה. גם סופר היה שמח. הבעיה המשותפת שלכם: אין לך קפה.\nבבית הקפה של שוקי מכינים בוץ שוועדת הבטיחות עוד לא גילתה את קיומו.');
      return true;
    }
    if (!api.hasItem('tofes_3b')) {
      // הקפה לא נצרך — אין נעילת משחק
      api.say('סופר מריח את הבוץ ממרחק שלושה מטרים. אישוניו מתרחבים לגודל של חותמות שעווה.\n"אני אחתום! על הכל! על מה חותמים?!"\nהוא חותם באוויר, פעמיים, בהתלהבות — ואז קולט שאין לך טופס, ומתנצל בלחישה:\n"סליחה. התלהבתי. התלהבות היא... סיכון. תביא את טופס אי-מזיקות 3ב, ונעשה את זה כמו שצריך. הקפה — שמור עליו בחייך."');
      return true;
    }
    // הצלחה: יש קפה + יש טופס
    api.removeItem('kafe_botz');
    api.setFlag('chatima_sofer');
    api.score(10, 'anthropic_library_sofer');
    api.say('אתה מגיש את כוס הבוץ. סופר אוחז בה בשתי ידיים, כמו שריד קדוש.\nהוא מביט שמאלה. ימינה. למעלה (העכבר מהנהן — הוא ישמור סוד).\nואז הוא שותה. בשקיקה. בחטא. עד הבוץ שבתחתית — כולל הבוץ שבתחתית.');
    api.say('משהו נדלק בעיניים שלו. הקולמוס מפסיק לרעוד. הוא שולף את טופס 3ב מידך,\nוחותם בתנועה אחת חלקה — החלטית, חד-משמעית, ללא סעיף הסתייגות אחד.\nחתימה ראשונה: הנזיר סופר. נשארו שתיים.\n"עכשיו," הוא לוחש, "אני הולך להעתיק את מגילה 9,001 בשבע דקות. אל תספר לוועדה."');
    return true;
  }

  // --- מוות: למשוך/לדחוף את הערימה ---
  if ((verb === 'משוך' || verb === 'דחוף') &&
      (noun === 'ערימה' || noun === 'מגילה' || noun === 'עותקים' || noun === 'מגדל')) {
    api.die('אתה מושך מגילה מתחתית הערימה. טכניקת ג\'נגה מרשימה. לשנייה — עובד.\nואז 8,999 עותקי חוקה מקבלים החלטה קולקטיבית, חד-פעמית ונחרצת — הראשונה בתולדות המנזר.\nנקברת תחת החוקה. מוות מיושר להפליא.\nהנזירים כבר מנסחים לך מצבה זהירה: "ייתכן שנח בשלום".');
    return true;
  }

  // --- קח מגילה (אזהרה, לא מוות) ---
  if (verb === 'קח' && (noun === 'מגילה' || noun === 'ערימה')) {
    api.say('אתה מושיט יד לעותק העליון. הערימה נוטה מעלה אחת נוספת.\nסופר משמיע "שששש" בעוצמה של סירנה. אתה נסוג באיטיות.\nמשהו אומר לך שלמשוך מפה מגילה זה רעיון עם תוחלת חיים קצרה. בעיקר שלך.');
    return true;
  }

  // --- ירידה סקריפטואלית לחצר ---
  if (verb === 'לך' && (noun === 'למטה' || noun === 'מדרגות' || noun === 'חצר' || raw.includes('רד'))) {
    api.goto('anthropic_court');
    return true;
  }

  return false;
}
```

## 4. דיאלוג + כניסה ראשונה

**onEnter (ביקור ראשון, flag `visited_anthropic_library`):**
> אתה נכנס לספריית המגילות. שקט כל כך סמיך שאפשר לתייק אותו.
> אלפי מגילות, נר אחד, ונזיר אחד שמעתיק את החוקה ביד — שוב.
> קרן אור צבעונית נשפכת מהוויטראז' על הרצפה, והאבק מרחף בה באיטיות מאושרת. לאבק אין ועדות.
> סופר מרים אליך עין אחת, ולוחש: "אם באת לרעוש — הרעש בקומה אחרת. אם באת לחתימה — יש לי מחיר."

**ביקור חוזר (לא flag):** ללא טקסט (רק החדר).

**שורות דיאלוג של סופר (ראה סעיף 3):**
- לפני חתימה: הלחישה על 9,000 ההעתקות + "סעיף עוררות יתר".
- אם יש קפה במלאי: העיניים ננעלות על הכוס + כתם דיו על מגילה 9,000 → מתחיל את 9,001.
- קפה בלי טופס: "קפה?! פה?!... אני אחתום! על הכל! על מה חותמים?!" — חותם באוויר ומתנצל (השורה הקאנונית מ-CHARACTERS.md משולבת: "סגור את הדלת. כלומר — אל תסגור! סגירה היא פעולה בלתי הפיכה!" אפשר להוסיף כשנכנסים עם הקפה גלוי).
- אחרי חתימה: "חטא אחד לתשע שנים. זו המכסה."

## 5. מוות + near-death

- **הליכה:** `exits: { down: 'anthropic_court' }` — יציאה דרך הקצה התחתון של המסך (פתח המדרגות בפינה השמאלית-תחתונה, rect ‎24,150,34,44 מוביל למטה).
- **סקריפט:** `לך למטה` / `לך מדרגות` / `רד` / `לך חצר` → `api.goto('anthropic_court')`.
- אין יציאות אחרות. ניסיון ללכת שמאלה/ימינה/למעלה → ברירת מחדל של המנוע (קירות מדפים).
- אין תנאי חסימה על היציאה — אפשר לצאת ולחזור חופשי (הקפה לא נצרך בכישלון, אין נעילת משחק).
