## 3. Puzzle script — onCommand

ניקוד (סה"כ **25**, תואם manifest):
- `openai_stage_confetti` = **15** — ירי קונפטי מוצלח בזמן מחיאות כפיים.
- `openai_stage_key` = **10** — חטיפת המפתח בתוך ענן הקונפטי.

פריטים נדרשים: `konfeti` (openai_corridor), `tag_tzevet` (openai_expo, דרך `bama_access`).
פריט מוענק: `mafteach_2`. דגל תוצר: `ostage_key_taken`.
דגלי עבודה פנימיים: `ostage_clock` (מונה תורות), `ostage_cloud` (תור שבו נורה הקונפטי),
`ostage_warned` (אזהרת "קח מפתח" ישיר), `ostage_wait` (מונה המתנה).

**מנגנון מחיאות הכפיים (תור-מבוסס):** כל פקודה בחדר מקדמת שעון. מחזור של 5 תורות:
תורות 4–5 במחזור = חלון מחיאות כפיים. כשחלון נפתח — הקריין מכריז (הרמז הקולי).
ירי קונפטי בחלון = הצלחה. מחוץ לחלון = כמעט-מוות סנארקי (הקונפטי לא נצרך).
אחרי ירי מוצלח יש **3 תורות** לקחת את המפתח לפני שהענן שוקע (הקונפטי לא נצרך גם אז —
"נשאר מטען אחרון", אין נעילת משחק).

```js
onCommand(verb, noun, api, extra) {
  const raw = (extra && extra.raw) || '';

  // ===== שעון האולם — מתקדם בכל פקודה בחדר =====
  const clock = (api.getFlag('ostage_clock') || 0) + 1;
  api.setFlag('ostage_clock', clock);
  const phase = clock % 5;                 // 4 או 0 (=5) → חלון מחיאות כפיים
  const applause = (phase === 4 || phase === 0);
  const cloudTurn = api.getFlag('ostage_cloud') || 0;
  const cloudActive = cloudTurn > 0 && (clock - cloudTurn) <= 3;
  if (cloudTurn > 0 && !cloudActive) {
    api.setFlag('ostage_cloud', 0);
    if (!api.getFlag('ostage_key_taken')) {
      api.say('הקונפטי שוקע ברכות על הבמה. המפתח שוב חשוף לעיני כולם.\nלמזלך, בתחתית התותח מקרקש מטען אחרון. אחרון-אחרון. הקריין לא מתערב יותר.');
    }
  }
  // הכרזת חלון (הרמז הקולי מהעיצוב):
  if (applause && !api.getFlag('ostage_key_taken')) {
    api.say('המגיש: "...ברבעון שאחרי הבא!" — רעם מחיאות כפיים מגלגל את האולם.\nאלפיים זוגות ידיים, אפס זוגות עיניים על הבמה. חלון הזדמנויות, אם היה לך במקרה... משהו חגיגי.');
  }

  // ===== ירי תותח הקונפטי =====
  const cannonNoun = (noun === 'תותח' || noun === 'קונפטי' || noun === 'תותח קונפטי' ||
                      raw.includes('קונפטי') || raw.includes('תותח'));
  if ((verb === 'השתמש' || verb === 'ירה' || verb === 'הפעל') && cannonNoun) {
    if (api.getFlag('ostage_key_taken')) {
      api.say('הבמה כבר מכוסה קונפטי, המפתח כבר בדלי. לירות שוב זה סתם שוויץ.\nטוב, נו. גם הקריין היה יורה. אבל לא. תתאפק.');
      return true;
    }
    if (!api.hasItem('konfeti')) {
      api.say('אתה עושה תנועת "פּוּם" עם הידיים. מרשים. חבל שתותח הקונפטי האמיתי\nעדיין מתאבק במסדרון הבטיחות, בין מחסום 12 למחסום 13.');
      return true;
    }
    if (!applause) {
      api.say('אתה מרים את התותח. האולם שקט. שקט של 2,000 איש שמחכים למשפט הבא.\nעכשיו? ברצינות? כל עדשה באולם תצלם אותך, ואבטחת ההייפ כבר מיישרת אפודים.\nהקריין, בטובו: חכה למחיאות הכפיים. הן מגיעות כמו שעון. שעון של הייפ.');
      return true;
    }
    // הצלחה — ירי בתוך חלון מחיאות כפיים
    api.setFlag('ostage_cloud', clock);
    api.score(15, 'openai_stage_confetti');
    api.say('פּוּם.\nענן קונפטי בגודל של סבב גיוס מתפוצץ מעל הבמה. הקהל, שכבר מחא כפיים,\nעובר להתעלפות מאורגנת — "אפקטים! באמצע הקינוט! זה בטח שחרור אמיתי!!"\nהמגיש מצטלם. הסדרן מצטלם. אף אחד לא מסתכל על הכן.\nיש לך שלוש שניות של אנונימיות מנצנצת. זוז.');
    return true;
  }

  // ===== קח מפתח =====
  const keyNoun = (noun === 'מפתח' || noun === 'מפתח כיול' || noun === 'פיצר' ||
                   raw.includes('מפתח'));
  if (verb === 'קח' && keyNoun) {
    if (api.getFlag('ostage_key_taken')) {
      api.say('המפתח כבר במים האפורים של הדלי, מוסווה כמו צוללת. אל תוציא אותו פה, גאון.');
      return true;
    }
    if (cloudActive) {
      // ===== ההצלחה הגדולה =====
      api.setFlag('ostage_cloud', 0);
      api.setFlag('ostage_key_taken');
      api.removeItem('konfeti');
      api.addItem('mafteach_2', "מפתח כיול מס' 2",
        "כרום מבריק, אריזת פריוויו. על התווית: 'זמין בקרוב'. הוא בידך. האירוניה מושלמת.");
      api.score(10, 'openai_stage_key');
      api.say('אתה חוצה את הבמה בתוך הענן, מרים את הזכוכית כמו מכסה של סיר,\nולוקח את מפתח כיול מס\' 2. הוא נוחת בדלי השטיפה בפליק שקט ונעלם במים האפורים.\nעל הכן אתה משאיר, מנומס, את שלט ה-DO NOT SHIP. הוראה זו הוראה.');
      api.say('הקהל מריע. מריע! הם בטוחים שצפו בהדגמה חיה של "פיצ\'ר ההיעלמות".\nמישהו בשורה שלוש כבר כותב ת\'רד: "ראיתי את העתיד והוא דלי".\nהקריין מציין לפרוטוקול: זו הפעם הראשונה בהיסטוריה שמשהו נשלח מהבניין הזה. בתוך דלי.\nעכשיו רד מהבמה לפני שיבקשו הדרן.');
      return true;
    }
    // ניסיון ישיר, בלי ענן — כמעט-מוות ואז מוות
    if (!api.getFlag('ostage_warned')) {
      api.setFlag('ostage_warned');
      api.say('אתה מושיט יד אל הזכוכית. פלאש. פלאש-פלאש. אלפיים טלפונים ננעלים עליך.\nהמגיש קופא באמצע מילה ("ברבע—"). מהאגפים, שלושה אפודים זוהרים מתחילים לנוע.\nהקריין, בנדיבות חד-פעמית: יד. בכיס. עכשיו. אתה "צוות ניקיון" — תתנהג כמו ענן.\nרמז: קהל שמוחא כפיים לא מצלם, ואין דבר שמסתיר במה טוב יותר מחגיגה.');
      return true;
    }
    api.die('שוב?! אמרתי "ענן", לא "יד".\nהיד שלך נוגעת בזכוכית. סירנת הייפ מייללת. אבטחת ההייפ עוטפת אותך כמו NDA.\nאתה מוכרז בו-במקום כ"חוויית משתמש לא מורשית" ומועבר לצוות המשפטי של ההתלהבות,\nשמפרק אותך לפוטנציאל, מקטלג אותך כ"למידה", ומתייק.\nמועד שחרורך: coming soon. כמו כל השאר פה. ☠');
    return true;
  }

  // ===== המתנה על הבמה — מוות מדורג =====
  if (verb === 'חכה' || verb === 'המתן' || raw === 'חכה' || raw.includes('המתן')) {
    const w = (api.getFlag('ostage_wait') || 0) + 1;
    api.setFlag('ostage_wait', w);
    if (w === 1) {
      api.say('אתה עומד על הבמה ולא עושה כלום. המגיש מבחין בך. עיניו נדלקות.\nזו לא הבעה טובה. זו הבעה של אדם שמצא "מומנט אותנטי".');
      return true;
    }
    if (w === 2) {
      api.say('המגיש מחליק לעברך עם המיקרופון: "וזה... איש תחזוקה אמיתי, חברים!"\nהקהל: "אווווו". זרקור מתחיל לזחול לכיוונך. הקריין ממליץ בחום: תזוז. תנקה. תירה משהו. כל דבר.');
      return true;
    }
    api.die('נשארת. בלי לזוז. על במת קינוט. מה חשבת שיקרה?\nהמגיש מניח יד על כתפך: "חברים — הפיצ\'ר הבא שלנו: יענקל\'ה!"\nהקהל משתגע. מסך ענק: התמונה שלך + "COMING SOON". שני סדרנים עוטפים אותך בניילון פריוויו\nומעבירים אותך למחסן ההשקות, בין GPT-6 לרובוט שמקפל כביסה.\nאתה נדחה לרבעון הבא. פיזית. לנצח. ☠');
    return true;
  }

  // ===== הקשב לקהל — רמז תזמון =====
  if ((verb === 'הקשב' || verb === 'שמע') ||
      (verb === 'הסתכל' && noun === 'קהל' && false /* look מנוהל ב-hotspot */)) {
    if (applause) {
      api.say('כרגע: רעם מחיאות כפיים. אלפיים איש, אפס עיניים על הבמה. זה הרגע מהסוג שמבזבזים.');
    } else {
      api.say('כרגע: דממת הייפ. הקהל סופג את המשפט הבא. לפי הקצב — עוד רגע המגיש יגיד\n"ברבעון שאחרי הבא", ואז יגיע הרעם. כמו שעון. שעון עם אופציות.');
    }
    return true;
  }

  // ===== ירידה סקריפטואלית =====
  if (verb === 'לך' && (noun === 'למטה' || noun === 'מדרגות' || noun === 'תצוגה' ||
      raw.includes('רד') || raw.includes('צא'))) {
    if (api.getFlag('ostage_key_taken')) {
      api.say('אתה מדרדר את העגלה במורד המדרגות. הסדרן מציץ בדלי: "מים מלוכלכים, אה?"\nאתה מהנהן. הוא מהנהן. המפתח שותק. שיתוף הפעולה המושלם.');
    }
    api.goto('openai_expo');
    return true;
  }

  return false;
}
```

**הערת יישום:** הכרזת חלון המחיאות (`api.say` בראש הפונקציה) רצה גם כשהפקודה לא
מטופלת בחדר (return false) — זה בכוונה: האולם חי גם כשאתה מקליד שטויות. אם המנוע
מציג את זה כפול/מציק — להעביר את ההכרזה רק לענפים המטופלים + ל'הקשב'.

## 4. דיאלוג + כניסה ראשונה

onEnter(api) {

## 5. מוות + near-death

**מוות 1 (נמנע — יש אזהרה מקדימה):** `קח מפתח` ישירות, פעם שנייה, בלי ענן קונפטי →
אבטחת ההייפ עוטפת אותך כמו NDA, אתה מפורק לפוטנציאל ומתויק.
> "מועד שחרורך: coming soon. כמו כל השאר פה. ☠"

**מוות 2 (נמנע — שתי אזהרות מדורגות):** `חכה` שלוש פעמים על הבמה → מוכרז כפיצ'ר
החדש, נעטף בניילון פריוויו ונדחה לרבעון הבא. פיזית. לנצח.

**Near-death gag:** ירי קונפטי שלא בזמן מחיאות כפיים → כל האולם מצלם אותך ואבטחת
ההייפ מיישרת אפודים; הקריין עוצר אותך שנייה לפני, בטובו ("חכה למחיאות הכפיים.
הן מגיעות כמו שעון. שעון של הייפ."). הקונפטי לא נצרך — אין נעילת משחק.

## 6. Exits

- **הליכה:** `exits: { down: 'openai_expo' }` — מדרגות הבמה בפינה הימנית-תחתונה
  (rect ‎288,150,32,30), יציאה דרך הקצה התחתון של המסך.
- **סקריפט:** `לך למטה` / `רד` / `צא` / `לך מדרגות` / `לך תצוגה` → `api.goto('openai_expo')`;
  אם `ostage_key_taken` — קודם שורת הדלי מול הסדרן (סעיף 3).
- **כניסה (מ-openai_expo, exit `enter`):** מותנית ב-`bama_access`. בלעדיו — הסדרן
  מחזיר אותך מיידית (`api.goto('openai_expo')` ב-onEnter, סעיף 4).
- אין יציאות נוספות. שמאלה/ימינה/למעלה → קירות האולם / ברירת מחדל של המנוע.
- אין חסימת יציאה אחרי הגניבה — הקהל בטוח שזו הייתה הדגמה; יוצאים עם הדלי בכבוד.
