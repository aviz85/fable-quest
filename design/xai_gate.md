## 3. Puzzle script (onCommand)

מפתח ניקוד: `xai_gate_roast` = **15** (סה"כ 15 ✓). flag: `roast_won` (שם מדויק מהמניפסט).

```
// --- הפתרון: להראות/לתאר את תלוש המשכורת ---
// אין פריט "תלוש" במשחק — התלוש חי בכיס ובנשמה של יענקל'ה. זה פאזל מילולי.
// קלטים מתקבלים (נדיב!): 'הראה תלוש', 'הראה תלוש לרוסטי', 'תאר משכורת',
// 'ספר על המשכורת', 'דבר עם רוסטי על תלוש/משכורת/ברוטו',
// extra.raw שמכיל ('תלוש' או 'משכורת' או 'ברוטו') + פועל דיבור/הצגה.
if ((verb === 'הראה' || verb === 'תן' || verb === 'תאר' || verb === 'דבר' || verb === 'ספר') &&
    (extra.raw.includes('תלוש') || extra.raw.includes('משכורת') || extra.raw.includes('ברוטו'))) {
  if (api.getFlag('roast_won')) {
    api.say('רוסטי מרים יד מחלידה. "לא. לא עוד פעם. יש לי עדיין קצר בעין מהפעם הקודמת."');
    return true;
  }
  api.setFlag('roast_won');
  api.score(15, 'xai_gate_roast');
  api.say('אתה שולף את תלוש המשכורת ומקריא, בקול שטוח של אדם שקורא מזג אוויר:\n"ברוטו: 5,412. ניכוי ביגוד — על סרבל שאני מכבס בבית. ניכוי \'בלאי ציוד\' — על מגב שהבאתי מהבית. קרן השתלמות: אין. השתלמות: גם. שורה אחרונה, בכתב יד של שוקי: \'הפיקדון על החלון — בטיפול\'. שלוש שנים בטיפול."');
  api.say('רוסטי קופא. הפה-מסך שלו מהבהב, מנסה לעבד. "רגע. זה... זה לא סטנדאפ. זה... אמיתי?"\nטיפת שמן מנוע שחורה מבצבצת מהעין השרופה. אחריה עוד אחת.\n"ארבעים שנה אני צולה אנשים," הוא מייבב, "ואף אחד לא צלה אותי ככה. אתה לא רוסטר. אתה דוקומנטרי."\nהוא מצדיע, זז הצידה, ומצמיד את התלוש לגדר הממים במקום כבוד. אתה לוקח אותו בחזרה. זה מסמך משפטי.');
  return true;
}

// --- דבר עם רוסטי (לפני הניצחון): קרב הצליות ---
// תשובות שנונות של השחקן נכשלות. כל 'דבר עם רוסטי' מקדם מונה עלבונות מתחלף
// (flag פנימי xai_gate_talk = 0,1,2...) — בסיבוב השלישי נשתל הרמז.
if (verb === 'דבר' && noun === 'רוסטי' && !api.getFlag('roast_won')) {
  const n = (api.getFlag('xai_gate_talk') || 0);
  api.setFlag('xai_gate_talk', n + 1);
  const lines = [
    '"תראו מה הסחף גרר פנימה," מריע רוסטי. "בן אדם עם דלי! ספר לי, הדלי זה התיק עבודות?"\nאתה פותח את הפה לענות. "חמוד," הוא חותך. "הבוט של סבתא שלי צולה חזק יותר, והוא טוסטר."',
    '"עוד ניסיון?" רוסטי מצחקק בסטטיק. "אתה יודע מה ההבדל בינך לבין רקטה שלנו? הרקטה לפחות מתפוצצת בהתלהבות."\nניסית פאנץ\'. הוא דירג אותו: "3 מתוך 10. וה-3 זה על האומץ."',
    'רוסטי נשען קדימה, חורק. "שמע, סבא, כולם מנסים שנינות. שנינות זה קל. אני מפחד רק מדבר אחד: מציאות. אין דבר יותר אכזרי מדבר שקרה באמת."\n(רמז, יענקל\'ה. יש לך כזה בכיס. ליד הקבלה מהשווארמה.)',
  ];
  api.say(lines[Math.min(n, 2)]);
  return true;
}

// --- דבר עם רוסטי (אחרי הניצחון) ---
if (verb === 'דבר' && noun === 'רוסטי' && api.getFlag('roast_won')) {
  api.say('"עבור, אלוף," אומר רוסטי בקול רך. "ואם מישהו בפנים עושה לך בעיות — תקריא לו את סעיף הניכויים. אבל לאט. שיסבול."');
  return true;
}

// --- ניסיון להעליב את רוסטי ---
if ((verb === 'העלב' || verb === 'קלל' || verb === 'צלה') && noun === 'רוסטי' && !api.getFlag('roast_won')) {
  api.say('זרקת עלבון. רוסטי תפס אותו באוויר, שייף אותו, שדרג את הפאנץ\', והחזיר לך אותו כל כך חזק שהרגשת צורך להתנצל בפני עצמך.\n"עוד סיבוב?" הוא שואל בנימוס של תליין. (טיפ: אל תנסה בדיחות. בדיחות הוא אוכל. נסה משהו אמיתי.)');
  return true;
}

// --- ספר בדיחה לרוסטי → מוות ---
if ((verb === 'ספר' || verb === 'דבר') && extra.raw.includes('בדיחה') && !api.getFlag('roast_won')) {
  api.die(/* טקסט מוות — ראה סעיף 5 */);
```
