## 3. Puzzle script (onCommand)

מפתח ניקוד: `anthropic_gate_open` = **15** (סה"כ 15 ✓)

```
// --- הצגת המגב לשער — הפתרון ---
if ((verb === 'השתמש' || verb === 'תן' || verb === 'הראה') &&
    (noun מכיל 'מגב') /* target: שער או בלי target */) {
  if (!api.hasItem('magav')) {
    api.say('היית שמח להראות מגב. אין לך מגב. יש לך רק חלומות ופיקדון שלא הוחזר.');
    return true;
  }
  if (api.getFlag('anthropic_gate_open')) {
    api.say('השער כבר פתוח. הוא מתרגש מחדש בכל פעם שהוא רואה את המגב, אבל בוא לא נסחט את הרגע.');
    return true;
  }
  api.setFlag('anthropic_gate_open');
  api.score(15, 'anthropic_gate_open');
  api.say('הרמת את המגב מול עיני השער.\nהשער פוקח את שתי העיניים לרווחה. "מגב... אדם עם מגב בא לנקות. ניקיון הוא הפחתת נזק. הפחתת נזק היא כוונה טהורה. אני... אני מורשה לשמוח?"\nהשער נפתח לאט, בחריקה שעברה בקרת עוצמה. "ברוך הבא, מפחית הנזקים. סליחה על החריקה. וסליחה על ההתנצלות."');
  return true;
}

// גם 'השתמש במגב' בלי יעד מפורש בחדר הזה → אותו דבר (fallthrough נדיב):
if (verb === 'השתמש' && noun === 'מגב') { /* כנ"ל */ }

// --- פתח/דחוף/משוך שער לפני הפתרון ---
if ((verb === 'פתח' || verb === 'דחוף' || verb === 'משוך') && noun === 'שער') {
  if (api.getFlag('anthropic_gate_open')) {
    api.say('השער פתוח. הוא מחזיק את עצמו פתוח בגאווה שקטה של דלת שסוף סוף מצאה משמעות.');
  } else {
    api.say('"אני מתנצל," אומר השער, "פתיחתי עלולה לגרום נזק. אי-פתיחתי עלולה לגרום נזק. אני בעיקר מקווה להיות משקוף."\nהוא נאנח. "הגשתי בקשה להיות חלון. נדחתה. חלונות נשברים."');
  }
  return true;
}

// --- דבר עם השער ---
if (verb === 'דבר' && (noun === 'שער' || noun === 'עיניים')) {
  if (api.getFlag('anthropic_gate_open')) {
    api.say('"אני פתוח לשיחה," אומר השער, "וגם סתם פתוח. עדיין מתרגל. תודה על המגב. ספרתי לזוגי, הדלת האחורית."');
  } else {
    api.say('"שלום," אומר השער בזהירות. "לפני שנמשיך, עליי לציין שהשיחה הזו עלולה להוביל לציפיות, וציפיות עלולות להוביל לאכזבה, ואכזבה היא נזק רגשי. שנמשיך? אתה בטוח? אני לא."\nהוא ממצמץ. "אנשים עם כוונות טהורות תמיד מזוהים לפי הציוד שלהם, אתה יודע."');
    // ^ רמז: הציוד = המגב
  }
  return true;
}

// --- שאל את השער איך נכנסים ---
if (verb === 'דבר' && extra.raw מכיל 'כניסה'/'להיכנס') → אותה תגובה עם הרמז.

// --- טפס על חומה → מוות ---
if (verb === 'טפס' && (noun === 'חומה' || noun === 'קיר')) {
  api.die(טקסט המוות — ראה סעיף 5);
  return true;
}

// --- טפס על שער ---
if (verb === 'טפס' && noun === 'שער') {
  api.say('התחלת לטפס. השער אומר בשקט: "אני חייב ליידע אותך שאני מגרד." אתה יורד. לא בגלל הגירוד — בגלל הטון.');
  return true;
}

// --- נגב חול / השתמש במגב על חול ---
if (verb === 'השתמש' && noun === 'מגב' && target === 'חול') {
  api.say('כמעט העברת מגב על גן הסלעים המגורף. השער פולט צליל שאפשר לתאר רק כ"התקף חרדה ביפנית". עצרת בזמן. חכם.');
  return true;
}

// --- לך ימינה / צא / היכנס למנזר בלי דגל ---
// מטופל ב-Exits (סעיף 6) — חסימה מנומסת.
```

הערת מימוש: לקבל גם `verb==='תן', noun==='מגב'` וגם `extra.raw.includes('מגב') && extra.raw.includes('שער')` — שחקנים ינסחו את זה בעשרים צורות. המגב **לא** נמסר לשער (לא `removeItem`) — יענקל'ה לא נפרד מהמגב לעולם.

> אתה עומד מול מנזר החוקה של אנת'רופיק. חומות חימר, גן סלעים מגורף, ושקט כל כך זהיר שאפילו הציקדות מגישות בקשה לצרצר.

## 5. Death — כרית האוויר החוקתית

**טריגר:** `טפס על חומה` / `טפס על קיר` (נמנע לחלוטין — פשוט אל תטפס).

> ☠ התחלת לטפס על החומה. החומה, שאומנה מיום היווסדה לא לגרום נזק לאף אדם, נכנסת לפאניקה: אתה עלול ליפול!
> תוך שנייה וחצי היא עוטפת אותך בכרית אוויר חוקתית — רכה, תומכת, ואטומה לחלוטין.
> "נשימה," מסבירה החומה בעודך מאבד הכרה, "היא פעולה עם פרופיל סיכון. אנחנו שומרים עליך ממנה."
> מתת בבטחה המלאה ביותר שתועדה אי פעם. שלוש מתוך שלוש ועדות מרוצות. [שחזר]

**Near-death gag (בונוס):** `אכול חול` →
> "בלעת חופן חול מגורף. השער מזעיק את ועדת החירום. עד שהיא תתכנס (רבעון שלישי), החול כבר יצא בדרך הטבע. שרדת, אבל הכבוד לא."
