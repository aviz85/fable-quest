```js
  if (verb === 'השתמש' && noun === 'מגב' && extra && /רצפה|קונפטי/.test(extra.raw || '')) {
    if (/קונפטי/.test(extra.raw || '')) {
      api.say("העברת מגב על הקונפטי. הרצפה נקייה. צפצוף: 'שינוי לא מאושר בסביבת העבודה — נפתח כרטיס'.\nבפעם הראשונה בקריירה שלך, ניקית משהו והרגשת עבריין. תחושה חדשה. לא נעימה. קצת ממכרת.");
    } else {
      api.say('העברת מגב על רצפה מבריקה. סתם, בשביל הכבוד המקצועי. הרצפה לא צריכה אותך.\nזה הדבר הכי עצוב שקרה לך השבוע, וזה שבוע שכולל חתימה על NDA כדי לנשום.');
    }
    return true;
  }

  // --- יציאות סקריפטיות ---
  if (verb === 'לך' && extra && /לובי|ימינה|חזרה|קבלה/.test(extra.raw || '')) {
    api.goto('openai_lobby');
    return true;
  }
  if (verb === 'לך' && extra && /תצוגה|שמאלה|במה|קדימה/.test(extra.raw || '')) {
    api.goto('openai_expo');
    return true;
  }

  return false;
}
```
